# Web app (server.py facade + backend/, desktop_app.py) — primary entry points
flask
pywebview
# Used in-process for both /api/check and /api/download (see
# backend/extraction.py / backend/download.py) - the vendored ./yt-dlp binary
# is no longer used by any code path.
yt-dlp

# yt-dlp's TikTok extractor (and others with heavy anti-bot measures) uses
# this to impersonate a real Chrome's TLS/HTTP fingerprint. Without it
# installed, yt-dlp logs "attempting impersonation, but no impersonate
# target is available" and TikTok extraction is far more likely to be
# blocked/challenged - confirmed live (2026-07-09): a real TikTok video hit
# an internal yt-dlp parsing crash without curl_cffi installed.
curl_cffi

# Primary resolver for Instagram profile/reels listings (yt-dlp's
# instagram:user extractor is blocked by Meta) - see backend/instagram.py.
instaloader
requests

# Auto-extracts a logged-in Instagram sessionid straight from the local
# browsers' cookie DBs (decrypts the macOS Keychain itself) so users don't have
# to export cookies.txt by hand - see backend/cookies.py's
# cookiefiles_from_browsers(). A separate implementation from yt-dlp's
# --cookies-from-browser, used as a more-reliable primary path plus a fallback
# when Keychain blocks yt-dlp.
browser_cookie3

# backend/paths.py points Python's global SSL_CERT_FILE at this package's
# bundled CA list at startup - a PyInstaller-frozen Python built inside a
# Homebrew/conda env otherwise bakes in a cert search path that only exists
# on the BUILD machine, breaking every raw urllib HTTPS call (Instagram/
# Threads/LinkedIn resolvers) on a real end-user machine. Was already an
# indirect dependency (yt-dlp/requests pull it in), but code now imports it
# directly, so it needs to be a real requirement, not an implicit one - see
# the curl_cffi entry above for why that distinction matters here.
certifi

# Testing (tests/test_server.py)
pytest

# Packaging the standalone macOS .app (see OmniFlow.spec / rules/packaging.md).
# Build-time only — not needed to run server.py/desktop_app.py from source.
pyinstaller
